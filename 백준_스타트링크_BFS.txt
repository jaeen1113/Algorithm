#include<iostream>
#include <cstring>
#include <queue>
using namespace std;
#pragma warning(disable:4996)

int f, s, g, u, d;
int check[1000001];
queue<int> q;
int bfs() {
	q.push(s);
	int click = 1;
	
	while (!q.empty()) {
		int temp = q.front(); q.pop();
		
		if (temp == g) break;

		int up = temp + u;
		int down = temp - d;

		if (up <= f && check[up] == 0) // 방문 안한 층일 떄만 
		{
			q.push(up);
			check[up] = check[temp] + 1;
		}
			
		
		if (down > 0 && check[down] == 0)
		{
			q.push(down);
			check[down] = check[temp] + 1;
		}
	}
	return check[g];
}
int main() {
	freopen("sample_input.txt" , "r", stdin);
	cin >> f >> s >> g >> u >> d;
	memset(check, 0, sizeof(check));
	int ret = bfs();

	if (ret != 0)
		cout << ret << endl;
	else
		cout << "use the stairs" << endl;

}
