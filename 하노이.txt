#include <stdio.h>
int N;
void solve(int N, int x, int y) { // 1부터 N개를 x(1)에서 y(3)로 옮긴다.
   if (N == 0) return;
   int z = 6 - x - y;
   solve(N - 1, x, z); // 1부터 N-1개를 1->2로 옮긴다.
   printf("%d %d\n", x, y); // N을 1에서 3으로 옮긴다.
   solve(N - 1, z, y); 
   
}
int main() {
   scanf("%d", &N);
   printf("%d\n", (1<<N) - 1);
   solve(N, 1, 3);
}
