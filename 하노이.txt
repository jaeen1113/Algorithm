#include<iostream>
#include<vector>
using namespace std;

void print(vector<vector<int> > *answer, int from, int to)
{
   vector<int> element = { from,to };
   cout << from<<to << "\n";
   answer->push_back(element);
}
void hanoi_recur(int n, vector<vector<int> > *answer, int from, int by, int to)
{
   //printf("from : %d   by : %d   to : %d\n",from,by,to);
   if (n == 1)
   {
      print(answer, from, to);
   }
   else
   {
      hanoi_recur(n - 1, answer, from, to, by);
      print(answer, from, to);
      hanoi_recur(n - 1, answer, by, from, to);
   }
}

vector<vector<int> > hanoi(int no)
{
   vector<vector<int> > answer;
   hanoi_recur(no, &answer, 1, 2, 3);
   return answer;
}

int main()
{
   int testNo = 3;

   vector<vector<int> > testAnswer = hanoi(testNo);

   // 아래는 테스트로 출력해 보기 위한 코드입니다.
   for (int i = 0; i< testAnswer.size(); i++)
   {
      for (int j = 0; j<testAnswer[i].size(); j++)
      {
         cout << testAnswer[i][j] << " ";
      }
      cout << "\n";
   }
   return 0;
}
