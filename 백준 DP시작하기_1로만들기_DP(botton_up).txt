#include <iostream>
#include <vector>
#include <cstring>
#include <algorithm>
#pragma warning(disable:4996)

using namespace std;
int dp[10000001];
int min2(int a, int b) {
	return a > b ? b : a; 
}
int make_one(int n) {
	dp[1] = 0;
	for (int i = 2; i <= n; i++) {
		dp[i] = dp[i - 1] + 1; // »©±â
		if (i % 2 == 0)
			dp[i] = min2(dp[i], dp[i / 2] + 1);
		if (i % 3 == 0)
			dp[i] = min2(dp[i], dp[i / 3] + 1);
	}

	return dp[n];

}
int main() {
	int n;
	cin >> n;
	memset(dp, -1, sizeof(dp));

	cout << make_one(n);
}
